	\subsection*{Pregunta 3 (Recursivo) }
	Recibe:
	\begin{itemize}
		\item m = tamaño de A
		\item n= tamaño de B
	\end{itemize}
	Devuelve: 
	\begin{itemize}
		\item 	Peso del min-matching
	\end{itemize}
	\begin{algorithmic}[1]
		\TITLE{\textsc{Min-Matching-Recursivo}$(A, B, a, b)$}
		\algcost{\textit{cost}}{\textit{times}}
		\STATE A = [a, ... , m]\algcost{$.$}{$.$}
		\STATE B = [b, ... , n]\algcost{$.$}{$.$}
		\IF{m = 1 \algcost{$.$}{$.$}}  
		\STATE return $A_1 / \sum_{x=1}^{n}B_x$ \algcost{$.$}{$.$}
		\ENDIF 
		\IF{n = 1 \algcost{$.$}{$.$}} 
		\STATE return $\sum_{x=1}^{m}A_x / B_1$ \algcost{$.$}{$.$}
		\ENDIF 
		\STATE vector$<$int$>$ posibles \algcost{$.$}{$.$}
		\FOR{$i=1$ TO $m-1$ \algcost{$.$}{$.$}}
		\STATE p = $\sum_{x=1}^{i}A_x/ B_1 + $ \\
		$\text{Min-Matching-Recursivo}(A, B, i+1, 2)$ \algcost{$.$}{$.$}
		\STATE posibles.push\_back(p) \algcost{$.$}{$.$}
		\ENDFOR
		\FOR{$i=1$ TO $n-1$ \algcost{$.$}{$.$}}
		\STATE p = $A_1 / \sum_{x=1}^{i}B_x + $ \\
		$\text{Min-Matching-Recursivo}(A, B, 2, i+1)$ \algcost{$.$}{$.$}
		\STATE posibles.push\_back(p) \algcost{$.$}{$.$}
		\ENDFOR
		\STATE return min(posibles)  \algcost{$.$}{$.$}
	\end{algorithmic}

	\subsection*{Pregunta 4 (Memoizado)}
		Recibe:
		\begin{itemize}
			\item m = tamaño de A
			\item n= tamaño de B
			\item Matriz[m][n]
		\end{itemize}
		Devuelve: 
		\begin{itemize}
			\item 	Peso del min-matching
		\end{itemize}
	\begin{algorithmic}[1]
		\TITLE{\textsc{Min-Matching-Memoizado}$(A, B, a, b)$}
		\algcost{\textit{cost}}{\textit{times}}
		\IF{Matriz[a][b] 	\algcost{$.$}{$.$}}
		\STATE return Matriz[a][b]	\algcost{$.$}{$.$}
		\ENDIF
		\STATE A = [a, ... , m]\algcost{$.$}{$.$}
		\STATE B = [b, ... , n]\algcost{$.$}{$.$}
		\IF{m = 1 \algcost{$.$}{$.$}}  
		\STATE return $A_1 / \sum_{x=1}^{n}B_x$ \algcost{$.$}{$.$}
		\ENDIF 
		\IF{n = 1 \algcost{$.$}{$.$}} 
		\STATE return $\sum_{x=1}^{m}A_x / B_1$ \algcost{$.$}{$.$}
		\ENDIF 
		\STATE vector$<$int$>$ posibles \algcost{$.$}{$.$}
		\FOR{$i=1$ TO $m-1$ \algcost{$.$}{$.$}}
		\STATE p = $\sum_{x=1}^{i}A_x/ B_1 + $ \\
		$\text{Min-Matching-Memoizado}(A, B, i+1, 2)$ \algcost{$.$}{$.$}
		\STATE posibles.push\_back(p) \algcost{$.$}{$.$}
		\ENDFOR
		\FOR{$i=1$ TO $n-1$ \algcost{$.$}{$.$}}
		\STATE p = $A_1 / \sum_{x=1}^{i}B_x + $ \\
	  	$\text{Min-Matching-Memoizado}(A, B, 2, i+1)$ \algcost{$.$}{$.$}
		\STATE posibles.push\_back(p) \algcost{$.$}{$.$}
		\ENDFOR
		\STATE Matriz[a][b] = min(posibles)
		\STATE return Matriz[a][b]  \algcost{$.$}{$.$}
	\end{algorithmic}
